Boolean And None


    print(False == 0) # True
    print(True == 1) # True
      
    print(True + True + True) # 3
    print(True + False + False) # 1
      
    print(None == 0) # False
    print(None == []) # False

Is and In


    print("a" in "apple") # True
    for i in "apple": 
        print(i) # "a", "p", "p", "l", "e"
    print(' ' is ' ') # True: string is immutable
    print({} is {}) # False: dictionary is mutable

Return and Yield


    def fun():
        S = 0
        for i in range(10):
            S += i
        return S
    print(fun()) # 45
      
    def fun():
        S = 0
        for i in range(10):
            S += i
            yield S
    for i in fun():
        print(i) # 0, 1, 3, 6, 10, 15, 21, 28, 36, 45

Global and nonlocal


    # global variable
    x = 15
    
    # nonlocal keyword
    def fun():
      var1 = 25
      def gun():
        nonlocal var1 # tells python explicitly to use var1 initialize in fun()
        global x # tells python explicitly to use x initialized globally
        var1 = var1 + x
        print(var1) # 40
      gun()
    fun()
